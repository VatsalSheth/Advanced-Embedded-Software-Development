114 boole gps_encode(char c)
115 {
116  boole valid_sentence = false;
117 
118 #ifndef GPS_NO_STATS
119  _encoded_characters++;
120 #endif
121  switch(c)
122  {
123  case ',': // term terminators
124  _parity ^= c;
125  case '\r':
126  case '\n':
127  case '*':
128  if (_term_offset < sizeof(_term))
129  {
130  _term[_term_offset] = 0;
131  valid_sentence = gps_term_complete();
132  }
133  ++_term_number;
134  _term_offset = 0;
135  _is_checksum_term = c == '*';
136  return valid_sentence;
137 
138  case '$': // sentence begin
139  _term_number = 0;
140  _term_offset = 0;
141  _parity = 0;
142  _sentence_type = GPS_SENTENCE_OTHER;
143  _is_checksum_term = false;
144  _is_gps_data_good = false;
145  return valid_sentence;
146  }
147 
148  // ordinary characters
149  if (_term_offset < sizeof(_term) - 1)
150  _term[_term_offset++] = c;
151  if (!_is_checksum_term)
152  _parity ^= c;
153 
154  return valid_sentence;
155 }

197 unsigned long gps_parse_decimal()
198 {
199  char *p;
200  boole isneg;
201  unsigned long ret;
202 
203  p = _term;
204  isneg = (*p == '-');
205  if (isneg)
206  ++p;
207 
208  ret = 100UL * gpsatol(p);
209 
210  while (gpsisdigit(*p))
211  ++p;
212 
213  if (*p == '.')
214  {
215  if (gpsisdigit(p[1]))
216  {
217  ret += 10 * (p[1] - '0');
218  if (gpsisdigit(p[2]))
219  ret += p[2] - '0';
220  }
221  }
222  return isneg ? -ret : ret;
223 }

228 unsigned long gps_parse_degrees()
229 {
230  char *p;
231  unsigned long left;
232  unsigned long tenk_minutes;
233 
234  left = gpsatol(_term);
235  tenk_minutes = (left % 100UL) * 10000UL;
236 
237  for (p=_term; gpsisdigit(*p); ++p);
238 
239  if (*p == '.')
240  {
241  unsigned long mult = 1000;
242  while (gpsisdigit(*++p))
243  {
244  tenk_minutes += mult * (*p - '0');
245  mult /= 10;
246  }
247  }
248  return (left / 100) * 100000 + tenk_minutes / 6;
249 }

257 boole gps_term_complete()
258 {
259  if (_is_checksum_term)
260  {
261  byte checksum;
262  checksum = 16 * from_hex(_term[0]) + from_hex(_term[1]);
263  if (checksum == _parity)
264  {
265  if (_is_gps_data_good)
266  {
267 #ifndef GPS_NO_STATS
268  ++_good_sentences;
269 #endif
270  _last_time_fix = _new_time_fix;
271  _last_position_fix = _new_position_fix;
272 
273  switch(_sentence_type)
274  {
275  case GPS_SENTENCE_GPRMC:
276  _time = _new_time;
277  _date = _new_date;
278  _latitude = _new_latitude;
279  _longitude = _new_longitude;
280  _speed = _new_speed;
281  _course = _new_course;
282  break;
283  case GPS_SENTENCE_GPGGA:
284  _altitude = _new_altitude;
285  _time = _new_time;
286  _latitude = _new_latitude;
287  _longitude = _new_longitude;
288  _numsats = _new_numsats;
289  _hdop = _new_hdop;
290  break;
291  }
292 
293  return true;
294  }
295  }
296 
297 #ifndef GPS_NO_STATS
298  else
299  ++_failed_checksum;
300 #endif
301  return false;
302  }
303 
304  // the first term determines the sentence type
305  if (_term_number == 0)
306  {
307  if (!gpsstrcmp(_term, GPRMC_TERM))
308  _sentence_type = GPS_SENTENCE_GPRMC;
309  else if (!gpsstrcmp(_term, GPGGA_TERM))
310  _sentence_type = GPS_SENTENCE_GPGGA;
311  else
312  _sentence_type = GPS_SENTENCE_OTHER;
313  return false;
314  }
315 
316  if (_sentence_type != GPS_SENTENCE_OTHER && _term[0])
317  switch(COMBINE(_sentence_type, _term_number))
318  {
319  case COMBINE(GPS_SENTENCE_GPRMC, 1): // Time in both sentences
320  case COMBINE(GPS_SENTENCE_GPGGA, 1):
321  _new_time = gps_parse_decimal();
322  _new_time_fix = uptime();
323  break;
324  case COMBINE(GPS_SENTENCE_GPRMC, 2): // GPRMC validity
325  _is_gps_data_good = (_term[0] == 'A');
326  break;
327  case COMBINE(GPS_SENTENCE_GPRMC, 3): // Latitude
328  case COMBINE(GPS_SENTENCE_GPGGA, 2):
329  _new_latitude = gps_parse_degrees();
330  _new_position_fix = uptime();
331  break;
332  case COMBINE(GPS_SENTENCE_GPRMC, 4): // N/S
333  case COMBINE(GPS_SENTENCE_GPGGA, 3):
334  if (_term[0] == 'S')
335  _new_latitude = -_new_latitude;
336  break;
337  case COMBINE(GPS_SENTENCE_GPRMC, 5): // Longitude
338  case COMBINE(GPS_SENTENCE_GPGGA, 4):
339  _new_longitude = gps_parse_degrees();
340  break;
341  case COMBINE(GPS_SENTENCE_GPRMC, 6): // E/W
342  case COMBINE(GPS_SENTENCE_GPGGA, 5):
343  if (_term[0] == 'W')
344  _new_longitude = -_new_longitude;
345  break;
346  case COMBINE(GPS_SENTENCE_GPRMC, 7): // Speed (GPRMC)
347  _new_speed = gps_parse_decimal();
348  break;
349  case COMBINE(GPS_SENTENCE_GPRMC, 8): // Course (GPRMC)
350  _new_course = gps_parse_decimal();
351  break;
352  case COMBINE(GPS_SENTENCE_GPRMC, 9): // Date (GPRMC)
353  _new_date = gpsatol(_term);
354  break;
355  case COMBINE(GPS_SENTENCE_GPGGA, 6): // Fix data (GPGGA)
356  _is_gps_data_good = (_term[0] > '0');
357  break;
358  case COMBINE(GPS_SENTENCE_GPGGA, 7): // Satellites used (GPGGA)
359  _new_numsats = (unsigned char)atoi(_term);
360  break;
361  case COMBINE(GPS_SENTENCE_GPGGA, 8): // HDOP
362  _new_hdop = gps_parse_decimal();
363  break;
364  case COMBINE(GPS_SENTENCE_GPGGA, 9): // Altitude (GPGGA)
365  _new_altitude = gps_parse_decimal();
366  break;
367  }
368 
369  return false;
370 }

405 float gps_distance_between (float lat1, float long1, float lat2, float long2)
406 {
407  // returns distance in meters between two positions, both specified
408  // as signed decimal-degrees latitude and longitude. Uses great-circle
409  // distance computation for hypothetical sphere of radius 6372795 meters.
410  // Because Earth is no exact sphere, rounding errors may be up to 0.5%.
411  // Courtesy of Maarten Lamers
412  float delta = radians(long1-long2);
413  float sdlong = (float)sin(delta);
414  float cdlong = (float)cos(delta);
415  lat1 = radians(lat1);
416  lat2 = radians(lat2);
417  float slat1 = sin(lat1);
418  float clat1 = cos(lat1);
419  float slat2 = sin(lat2);
420  float clat2 = cos(lat2);
421  delta = (clat1 * slat2) - (slat1 * clat2 * cdlong);
422  delta = sq(delta);
423  delta += sq(clat2 * sdlong);
424  delta = sqrt(delta);
425  float denom = (slat1 * slat2) + (clat1 * clat2 * cdlong);
426  delta = atan2(delta, denom);
427  return delta * 6372795;
428 }

438 float gps_course_to (float lat1, float long1, float lat2, float long2)
439 {
440  // returns course in degrees (North=0, West=270) from position 1 to position 2,
441  // both specified as signed decimal-degrees latitude and longitude.
442  // Because Earth is no exact sphere, calculated course may be off by a tiny fraction.
443  // Courtesy of Maarten Lamers
444  float dlon = radians(long2-long1);
445  lat1 = radians(lat1);
446  lat2 = radians(lat2);
447  float a1 = sin(dlon) * cos(lat2);
448  float a2 = sin(lat1) * cos(lat2) * cos(dlon);
449  a2 = cos(lat1) * sin(lat2) - a2;
450  a2 = atan2(a1, a2);
451  if (a2 < 0.0)
452  {
453  a2 += TWO_PI;
454  }
455  return degrees(a2);
456 }
